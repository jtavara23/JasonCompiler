Program ex5;
variables
	integer m;
function f returns integer;
	parameters
		integer n;
		var integer k;
	variables
		integer p,q,saida;
	begin
		if n < 2 then
			set k=0;
			set saida=n
		else
			set k=p+q+1;
			set saida = f(n-1,p)+f(n-2,q)
		endif;
		return (saida)
	end;
begin
	write f(3,m);
	write m
end.
